    height = canvas.height;

  function disturb(gx, gy, radius = 2, strength = 3) {
    for (let y = -radius; y <= radius; y++) {
      const py = gy + y;
      if (py < 1 || py >= rows - 1) continue;
      for (let x = -radius; x <= radius; x++) {
        const px = gx + x;

  function disturbDirectional(gx, gy, vx, vy, trailLen = 10, strength = 3) {

      const py = Math.round(gy - vy * s);

    const gx = Math.floor(x / gridScale);

      lastGX = gx;
      lastGY = gy;
      return;

    const dx = gx - lastGX;
    const dy = gy - lastGY;
    const dist = Math.sqrt(dx * dx + dy * dy);

    if (dist === 0) return;

    const vx = dx / dist;   // å•ä½æ–¹å‘å‘é‡

  function updateHeightField() {

      const upOffset = (y - 1) * cols;
      const downOffset = (y + 1) * cols;
      for (let x = 1; x < cols - 1; x++) {
        const i = x + yOffset;
        const left = i - 1;
        const right = i + 1;
        const up = x + upOffset;
        const down = x + downOffset;

    const tmp = prev;
    prev = curr;

  function lerp(a, b, t) {

  function render() {
    const img = ctx.getImageData(0, 0, width, height);
    const data = img.data;

    // èƒŒæ™¯è‰²ï¼ˆå’Œ CSS é‡Œä¿æŒä¸€è‡´ï¼‰
    const base = { r: 100, g: 150, b: 150 };

    // å…ˆæ•´ä½“å¡«å……èƒŒæ™¯
    for (let i = 0; i < data.length; i += 4) {

      data[i + 2] = base.b;
      data[i + 3] = 255;
